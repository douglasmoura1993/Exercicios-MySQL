1. SELECIONE TODOS OS DIRETORES QUE NÃO POSSUEM FILMES RELACIONADOS:

SELECT ATOR_DIRETOR.ID_ATORD, NOME_ATORD, NAC_ATORD, FILME.ID_ATORD
FROM ATOR_DIRETOR LEFT JOIN FILME ON
ATOR_DIRETOR.ID_ATORD = FILME.ID_ATORD 
WHERE FILME.ID_ATORD IS NULL;

2. SELECIONE OS FILMES E SEUS RESPECTIVOS ATORES:

SELECT FILME.ID_ATORD, TIT_PT_FILME, ID_FILME, CLASSIF_FILME, ATOR_DIRETOR.ID_ATORD, NOME_ATORD
FROM FILME LEFT JOIN ATOR_DIRETOR ON
FILME.ID_ATORD = ATOR_DIRETOR.ID_ATORD;

3. SELECIONE OS GÊNEROS QUE NÃO ESTÃO RELACIONADOS A NENHUM FILME:

SELECT NOME_GENERO, GENERO.ID_GEN_FILME, FILME.ID_GEN_FILME
FROM GENERO LEFT JOIN FILME ON 
GENERO.ID_GEN_FILME = FILME.ID_GEN_FILME
WHERE FILME.ID_GEN_FILME IS NULL;

4. SELECIONE OS CINEMAS E SEUS RESPECTIVOS FILMES:

SELECT NOME_CIN AS CINEMA, END_CIN AS ENDERECO, CIDADE_CIN AS CIDADE, CINEMA.ID_CIN, EXIBE.ID_CIN, EXIBE.ID_FILME, FILME.ID_FILME,
TIT_PT_FILME AS TITULO
FROM CINEMA LEFT JOIN EXIBE ON EXIBE.ID_CIN = CINEMA.ID_CIN
LEFT JOIN FILME ON FILME.ID_FILME = EXIBE.ID_FILME;

5. SELECIONE OS FILMES QUE NÃO ESTÃO EM NENHUMA SESSÃO:

SELECT EXIBE.ID_SESSAO, EXIBE.ID_FILME, FILME.ID_FILME, TIT_PT_FILME
FROM EXIBE LEFT JOIN FILME ON 
EXIBE.ID_FILME = FILME.ID_FILME
WHERE EXIBE.ID_SESSAO IS NULL;
